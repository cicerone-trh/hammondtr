1 ***********************************************************************
  2 * Calculator Program:
  3 *   Program takes in 7 characters of input to be used as three
  4 *   values, as follows:
  5 *       xxxayyy
  6 *   where x is the first numeric operand, a is the arithmetic symbol
  7 *   and y is the second numeric operand
  8 *
  9 *   Available functions include:
 10 *       Addition:       123$321 is 123 + 321
 11 *       Subtraction:    321-123 is 321 - 123
 12 *       Multiplication: 123*123 is 123 * 123
 13 *
 14 *   Output is in a very readable format:
 15 *       YOUR INPUT WAS 123$321
 16 *       123 PLUS 321 IS 444
 17 **********************************************************************
 18 *        10    16                 35
 19 ***********************************************************************
 20          BEGIN NAME=GTT0
 21          MI0MI REG=R3
 22          L     R3,CE1CR0
 23 ***********************************************************************
 24 *Check Length of input: must be 7 characters w/o action code
 25 ***********************************************************************
 26 INIT     L     R0,MI0CCT          Loading to check value
 27          CHI   R0,13              2+3+1+7 = 13
 28          BNE   NOT7               If incorrect input length
 29          LA    R12,MI0ACC+2       Load address of first input byte
 30 ***********************************************************************
 31 *Check that the first 3 bytes are a number
 32 *If not, display inproper operand message
 33 ***********************************************************************
 34          LHI   R0,3               Set counter for chknm
 35          L     R13,=X'F0'         Set minimum hex value
 36          L     R14,=X'F9'         Set maximum hex value
 37          LHI   R7,1               Switch for CHKNM2
 38 CHKNM    EQU   *                  Loop to chk numeric only
 39          CL    R13,0(R12)         Check if <0
 40          BL    NOTNUM             Display error message
 41          CL    R14,0(R12)         Check if <0
 42          BH    NOTNUM             Display error message
 43          LA    R12,1(R12)         Increase pointer to next byte
 44          BCTR  R0,CHKNM
 45 ***********************************************************************
 46 *Check that last 3 bytes are numeric
 47 *If not, display inproper operand message
 48 ***********************************************************************
 49          CHI   R7,1               If true, have not done 2nd loop
 50          BNE   SETOPS             If already done, continue to next
 51          LHI   R0,3               Sets counter
 52          LA    R12,MI0ACC+6       Sets pointer to second operand
 53          LHI   R7,0               Sets loop check switch to false
 54          B     CHKNM              Goes through loop with new R12
 55 ***********************************************************************
 56 *Places the numeric input values into registers R2 and R4
 57 ***********************************************************************
 58 SETOPS   EQU   *
 59          LHI   R0,0               Temporary usage
 60          ST    R0,EBW000          Sets first 4 bytes to 00
 61          ST    R0,EBW004          Sets second 4 byte to 00
 62          PACK  EBW005(3),MI0ACC+2 Prepare for CVB (005-008 are used)
 63          CVB   R12,EBW000         Places op1 in r12
 64          PACK  EBW005(3),MI0ACC+6 Prepare for R4 as operand two
 65          CVB   R14,EBW000         Places op2 in r14
 66          B     CHKOP              Onward to operator validation
 67 ***********************************************************************
 68 *Check that the arithmetic indicator character is valid*
 69 *If found, branch to that function
 70 *If not, display inproper arithmetic operator
 71 ***********************************************************************
 72 CHKOP    SR    R6,R6              R6 = operator
 73          IC    R6,MI0ACC+5        Loads operator as hex into r6
 74          CL    R6,=X'5B'          5B = '$'
 75          BE    ADDIT              If operator is *ADD*
 76          CL    R6,=X'60'          60 = '-'
 77          BE    SUBIT              If operator is *SUB*
 78          CL    R6,=X'5C'          5C  = '*'
 79          BE    MULTIT             If operator is *MULTIPLY*
 80          B     BADOP              If operator is none of the above
 81 ***********************************************************************
 82 *MATH FUNCTIONS: ADDIT (R12 + R14)
 83 ***********************************************************************
 84 ADDIT    EQU   *
 85          CVD   R12,EBW000                Begin process to print op1
 86          UNPK  EBW000(4),EBW004          op1 as character data
 87          MVC   EBW004(6),=C' PLUS '      Concatenate message
 88          CVD   R14,EBW010                Begin process to prnt op2
 89          UNPK  EBW010(4),EBW014          op2 as char data
 90          AR    R12,R14                   Performs addition operation
 91          MVC   EBW014(4),=C' IS '        Concatenate literal
 92          CVD   R12,EBW018                sum as char data
 93          UNPK  EBW018(4),EBW022          final part of message
 94          MVI   EBW022,#EOM               prepare for entnc
 95          SR    R2,R2
 96          ENTNC UIO1
 97 ***********************************************************************
 98 *MATH FUNCTIONS: SUBIT (R12 - R14)
 99 ***********************************************************************
100 SUBIT    EQU   *
101          CVD   R12,EBW000                Begin process to print op1
102          UNPK  EBW000(4),EBW004          op1 as character data
103          MVC   EBW004(7),=C' MINUS '       Concatenate message
104          CVD   R14,EBW011                Begin process to prnt op2
105          UNPK  EBW011(4),EBW015          op2 as char data
106          SR    R12,R14                   Performs subtraction operation
107          MVC   EBW015(4),=C' IS '        Concatenate literal
108          CVD   R12,EBW019                Diff as char data
109          UNPK  EBW019(4),EBW022          final part of message
110          MVI   EBW023,#EOM               prepare for entnc
111          SR    R2,R2
112          ENTNC UIO1
113 ***********************************************************************
114 *MATH FUNCTIONS: MULTIT (R12 * R14)
115 ***********************************************************************
116 MULTIT   EQU   *
117          CVD   R12,EBW000                Begin process to print op1
118          UNPK  EBW000(4),EBW004          op1 as character data
119          MVC   EBW004(7),=C' TIMES '     Concatenate message
120          CVD   R14,EBW011                Begin process to prnt op2
121          UNPK  EBW011(4),EBW015          op2 as char data
122          MLR   R12,R14                   multiply op
123          MVC   EBW015(4),=C' IS '        Concatenate literal
124          CVD   R12,EBW019                Diff as char data
125          UNPK  EBW019(4),EBW023          final part of message
126          MVI   EBW023,#EOM               prepare for entnc
127          SR    R2,R2
128          ENTNC UIO1
129 ***********************************************************************
130 *ERROR DISPLAY MESSAGES:
131 ***********************************************************************
132 NOT7     EQU   *
133          SENDA MSG='INPUT LENGTH WAS NOT 7'
134          B     ENDPGM
135 NOTNUM   EQU   *
136          CL    R12,MI0ACC+5
137          BL    NOTNUM1
138          BH    NOTNUM2
139 NOTNUM1  EQU   *
140          SENDA MSG='FIRST OP NOT NUMERIC'
141          B     ENDPGM
142 NOTNUM2  EQU   *
143          SENDA MSG='SECOND OP NOT NUMERIC'
144          B     ENDPGM
145 BADOP    EQU   *
146          SENDA MSG='OPERATOR NOT RECOGNIZED'
147          B     ENDPGM
148 ENDPGM   EQU   *
149          FINIS
150          END
